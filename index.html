class TechEntrepreneur:
    def __init__(self, name):
        self.name = name
        self.skills = ["Full-Stack Dev", "Business Strategy"]
        self.projects = []

    def learn(self, skill):
        self.skills.append(skill)
        print(f"Learned {skill}!")

    def build_startup(self, idea):
        startup = f"{idea} by {self.name}"
        self.projects.append(startup)
        return startup

# Your journey starts here
you = TechEntrepreneur("Your Name")
you.learn("AI Development")
new_startup = you.build_startup("NextGen AI Platform")
print(f"Launching: {new_startup}")
